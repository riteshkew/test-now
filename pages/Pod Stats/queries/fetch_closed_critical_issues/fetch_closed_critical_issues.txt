SELECT issue_number, title, 'Closed' as series, label_arr, to_char(closed_date, 'YYYY-MM') as full_date, to_char(closed_date, 'Mon') as month FROM github_issues where 'Critical' = ANY(label_arr) and '{{appsmith.store.currentPod}}' = ANY(label_arr) 
and 'Bug' = ANY(label_arr) and to_char(closed_date, 'YYYY-MM') >= '{{bug_start_select.selectedOptionValue}}' and to_char(closed_date, 'YYYY-MM') <= '{{bug_end_select.selectedOptionValue}}' GROUP BY id, full_date order by full_date,issue_number
